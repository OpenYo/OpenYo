CREATE DATABASE OpenYo;
USE OpenYo;
CREATE TABLE friends (userId VARCHAR(31) not null, friend VARCHAR(31) not null);
CREATE TABLE callback(userId VARCHAR(31) not null, uri    VARCHAR(255) not null);
CREATE TABLE apiToken (userId VARCHAR(31) not null, token  VARCHAR(63) not null);
CREATE TABLE imkayac (userId VARCHAR(31) not null, kayacId VARCHAR(31) not null, password VARCHAR(31), seckey VARCHAR(31));
CREATE TABLE yoUser (userId VARCHAR(31) not null, yoId VARCHAR(31) not null);
CREATE TABLE notifyType (userId VARCHAR(31) not null, type VARCHAR(31) not null);
CREATE TABLE password (userId VARCHAR(31) not null, salt VARCHAR(63) not null, hash CHAR(128) not null);
CREATE TABLE GCMApiToken (project VARCHAR(15) not null, token VARCHAR(63) not null);
CREATE TABLE GCMRegId (userId VARCHAR(31) not null, project VARCHAR(15) not null, regId VARCHAR(4096) not null);
CREATE TABLE logYoed (userId VARCHAR(31) not null, fromUser VARCHAR(31) not null, time DATETIME not null);
CREATE TABLE banWhen (userId VARCHAR(31) not null, forT INT not null, time DATETIME not null);
